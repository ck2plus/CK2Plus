# Feudal Group
feudal_group_potential_trigger = { # For values shared by multiple government types
	# These governments don't play by the rules
	trigger_if = {
		limit = { chinese_imperial_government_potential_trigger = yes }
		always = yes
	}
	trigger_else_if = {
		limit = { roman_imperial_government_potential_trigger = yes }
		always = yes
	}
	trigger_else_if = {
		limit = { muslim_imperial_government_potential_trigger = yes }
		always = yes
	}
	trigger_else_if = {
		limit = { order_government_potential_trigger = yes }
		always = yes
	}
	# The rest DO play by the rules
	trigger_else = {
		is_patrician = no
		# Keep Tribes and Nomads from instant-switching to Feudal/Republican
		is_tribal = no
		is_nomadic = no

		imperial_government_type_trigger = no
	}

	trigger_if = {
		limit = { controls_religion = yes }

		temporal_religious_head_trigger = yes
	}
}
feudal_group_allow_trigger = {}


# Chinese Imperial
chinese_imperial_government_potential_trigger = {
	trigger_if = {
		limit = {
			NOT = { has_character_flag = chinese_imperial_government_preserve } # keeps government when player is given culture change opton
		}

		OR = {
			culture_group = chinese_group
			culture = khitan
			culture = tangut
			culture = jurchen
		}
	}

	primary_title = {
		real_tier = EMPEROR
		has_title_flag = pretender_chinese_empire
	}
}

# Holy Orders

	trigger_if = {
		limit = { holy_order = no }

		ai = yes

		any_liege = {
			holy_order = yes
			religion = ROOT
		}

		NOT = { is_government_potential = merchant_republic_government }
	}
}

# Imperial Roman
roman_imperial_government_potential_trigger = {
	muslim_government_potential_trigger = no

	trigger_if = {
		limit = { independent = yes }

		trigger_if = {
			limit = {
				has_game_started = no
				is_save_game = no
			}
			primary_title = {
				OR = {
					title = e_roman_empire
					title = e_byzantium
				}
			}
		}
		trigger_else = {
			primary_title = {
				OR = {
					has_law = succ_byzantine_elective
					imperial_government_type_trigger = yes
				}
			}
		}
	}
	trigger_else = {
		top_liege = { roman_imperial_government_potential_trigger = yes }
	}
}
roman_imperial_government_allow_trigger = {}

# Merchant Republic
merchant_republic_government_potential_trigger = {
	# Patrician and Doge requirements
	OR = {
		# Doge requirements
		# Does not require being a patrician, to allow creation of new Merchant Republics
		AND = {
			# Must always be coastal
			capital_scope = { is_coastal = yes }

			# Require at least one province and be Duke tier or higher
			higher_real_tier_than = COUNT
			num_of_count_titles >= 1

			# MR can't be under MR
			trigger_if = {
				limit = { independent = no }

				liege_before_war = {
					is_merchant_republic = no
				}
			}
		}

		# Patrician requirements
		AND = {
			is_patrician = yes

			# Liege must be MR
			liege_before_war = {
				is_merchant_republic = yes

				# MR can't be under MR
				trigger_if = {
					limit = { independent = no }

					liege_before_war = {
						is_merchant_republic = no
					}
				}
			}
		}
	}
}
merchant_republic_government_allow_trigger = {}
