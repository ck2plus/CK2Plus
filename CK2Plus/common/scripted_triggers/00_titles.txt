# Triggers relating to landed titles

can_create_title = {
	is_title_active = FROM # Title has to be active to be creatable
	# empty for vanilla
}

can_create_duchy = {
	can_create_title = yes
	# empty for vanilla
}

can_create_kingdom = {
	can_create_title = yes
	# empty for vanilla
}

can_create_empire = {
	can_create_title = yes
	# empty for vanilla
}

can_usurp_religion_head = {
	num_of_holy_sites >= 3
	piety >= 500
}

has_cultural_titular_requirements = {
	trigger_if = {
		ai = yes
		culture = FROM
	}
}

has_titular_requirements = {
	FROM = {
		trigger_if = {
			limit = { is_titular = yes }
			OR = {
				has_holder = yes
				ROOT = { culture = FROM }
			}
		}
	}
}

muslim_religion_head_trigger = {
	NOT = { has_global_flag = shia_caliphate_revolt_ongoing }

	trigger_if = {
		limit = { has_global_flag = destroyed_caliphates }

		custom_tooltip = {
			text = controls_mecca_medina

			c_mecca = {
				owner = {
					OR = {
						character = ROOT
						is_vassal_or_below_of = ROOT
					}
				}
			}
			c_medina = {
				owner = {
					OR = {
						character = ROOT
						is_vassal_or_below_of = ROOT
					}
				}
			}
		}

		OR = {
			custom_tooltip = {
				text = controls_jerusalem_damascus_baghdad

				c_jerusalem = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
				c_damascus = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
				c_baghdad = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
			}

			AND = {
				trait = sayyid
				piety >= 1000
			}
		}
	}
	trigger_else = {
		OR = {
			custom_tooltip = {
				text = controls_jerusalem_damascus_baghdad

				c_jerusalem = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
				c_damascus = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
				c_baghdad = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
			}

			custom_tooltip = {
				text = controls_mecca_medina

				c_mecca = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
				c_medina = {
					owner = {
						OR = {
							character = ROOT
							is_vassal_or_below_of = ROOT
						}
					}
				}
			}

			AND = {
				trait = sayyid
				piety >= 1000
			}
		}
	}
}
