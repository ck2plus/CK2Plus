government_flavor = {
	name = gov_electorial_aristocratic_monarchy
	graphics = GFX_evt_found_hre
	priority = 1

	trigger = {
		OR = {
			government = feudal_government
			government = imperial_government
			government = bogomilist_government
			government = bogomilist_imperial_government
			government = monastic_feudal_government
		}

		OR = {
			has_law = succ_feudal_elective
			has_law = succ_hre_elective
		}

		religion_openly_norse_or_reformed_trigger = no
		religion_openly_german_or_reformed_trigger = no
		religion_openly_tengri_or_reformed_trigger = no

		NOR = {
			has_law = imperial_administration
			has_law = ze_administration_laws_2
		}

		trigger_if = {
			limit = { has_dlc = "Conclave" }

			calc_true_if = {
				amount >= 6

				has_law = law_voting_power_1
				has_law = banish_voting_power_1
				has_law = execution_voting_power_1
				has_law = revoke_title_voting_power_1
				has_law = grant_title_voting_power_1
				has_law = imprison_voting_power_1
				has_law = war_voting_power_1
			}
		}
	}
}
