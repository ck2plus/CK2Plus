			trigger_if = {

					has_law = crown_authority_4
				}
			}

			trigger_if = {
				limit = { has_dlc = "Conclave" }

				custom_tooltip = {
					text = forced_elective_tt

					NAND = {
						has_law = succession_voting_power_1
						OR = {
							has_law = succ_feudal_elective
							has_law = succ_hre_elective
						}
					}
				}
			}

			holder_scope = {
				show_scope_change = no

				NOT = { trait = uncrowned }

				trigger_if = {
					limit = { in_revolt = yes }

				}

				trigger_if = {
					limit = { is_tribal = yes }

					OR = {
						has_law = tribal_organization_3
						has_law = tribal_organization_4
					}
				}

				trigger_if = {
					limit = { independent = no }

					custom_tooltip = {
						text = not_forced_tt
						NOT = { has_character_flag = liege_forced_succ_law }
					}
				}

				trigger_if = {
					limit = { real_tier = COUNT }
					prestige >= 250
				}
				trigger_else_if = {
					limit = { real_tier = DUKE }
					prestige >= 500
				}
				trigger_else_if = {
					limit = { real_tier = KING }
					prestige >= 1000
				}
				trigger_else_if = {
					limit = { real_tier = EMPEROR }
					prestige >= 2000
				}
			}
		}

		pass_effect = {
			log = "succ_gavelkind pass ([This.GetFullBaseName])"
			if = {
				limit = { has_game_started = yes }

				holder_scope = {
					show_scope_change = no
					trigger_switch = {
						on_trigger = real_tier
						COUNT = { prestige = -250 }
						DUKE = { prestige = -500 }
						KING = { prestige = -1000 }
						EMPEROR = { prestige = -2000 }
					}
				}
			}
		}

		}
