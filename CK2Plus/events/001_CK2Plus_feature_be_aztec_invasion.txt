	desc = EVTDESC_plusSSI.001 # Write this later
	picture = GFX_evt_aztec_explorers

	is_triggered_only = yes # on_chronicle_start (move to on_startup?)
	ai = no

	trigger = {

		is_save_game = no # Gotta do it at the beginning or not at all
		multiplayer = no # Too much of a pain to design for multiplayer scenarios. May rework later

		religion = aztec_pagan
		culture = nahua

		any_demesne_province = {
			OR = {
				region = custom_north_sea_zone
				region = custom_atlantic_zone
			}

			is_coastal = yes
			owned_by = PREV
		}
	}

	immediate = { # This was in an Option originally. I think the chances of someone meeting all the conditions for the trigger and *not* wanting to play the invasion are low
		hidden_effect = {
			activate_title = {
				title = e_aztec_empire
				status = yes
			}
		}

		prestige = 500
		gain_title = e_aztec_empire

		any_demesne_title = {
			show_scope_change = no

			}
		}

		hidden_effect = {
			capital_holding = {
				if = {
					limit = { holding_type = tribal }
					convert_to = castle
				}
			}

			set_government_type = feudal_government

			e_aztec_empire = {
				set_title_flag = non_manual_law_change # No prestige loss
				add_law = crown_authority_2
				add_law = council_privileges_0
				add_law = succ_primogeniture
				clr_title_flag = non_manual_law_change # No prestige loss
			}

			set_global_flag = player_aztec
			set_global_flag = aztec_arrival
			set_global_flag = aztec_explorers

			random_demesne_province = {
				limit = { is_coastal = yes }
				preferred_limit = { is_capital = yes }

				save_global_event_target_as = first_aztec_invasion_target
			}
		}
	}

	option = {
		name = EVTOPTA_SSI_74
	}

	after = { narrative_event = { id = SSI.4 } }
}
