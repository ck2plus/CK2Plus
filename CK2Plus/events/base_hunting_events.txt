# A Majestic Stag
character_event = {
	id = 36000
	desc = EVTDESC36000
	picture = GFX_evt_hunt

	only_playable = yes
	has_character_flag = hunt_begin
	min_age = 16
	only_capable = yes
	prisoner = no

	trigger = {
		has_character_modifier = holding_grand_hunt

		trigger_if = {
			limit = { is_female = yes }

			OR = {
				trait = martial_lady
				trait = martial_lady_pagan
				trait = martial_lady_tengri
			}
		}

		had_character_flag = {
			flag = hunt_begin
			days >= 3
		}

		health_traits < 1

		trigger_if = {
			limit = { has_character_flag = hunt_event_happened }

			had_character_flag = {
				flag = hunt_event_happened
				days >= 3
			}
		}
	}

	immediate = {
		set_character_flag = hunt_event_happened
	}

	mean_time_to_happen = {
		days = 200

		mult_modifier = {
			factor = 0.33
			NOT = { has_character_flag = hunt_event_happened }
		}

		mult_modifier = {
			factor = 1.5
			has_character_flag = hunt_event_happened
		}

		mult_modifier = {
			factor = 1.5
			trait = content
		}

		mult_modifier = {
			factor = 0.8
			trait = hunter
		}

		mult_modifier = {
			factor = 5.0
			martial >= 15
		}

		mult_modifier = {
			factor = 2.0
			trait = slothful
		}

		mult_modifier = {
			factor = 0.5
			has_character_flag = hunt_good_hounds
		}

		mult_modifier = {
			factor = 1.25
			trait = craven
		}
	}

	option = {
		name = EVTOPTA36000

		trigger = {
			NOT = { trait = craven }
		}

		prestige = 5
	}

	option = {
		name = EVTOPTB36000
		tooltip_info = craven

		trigger = { trait = craven }

		prestige = -10
	}

	option = {
		name = EVTOPTC36000

		trigger = {
			NOR = {
				trait = greedy
				trait = arbitrary
				trait = envious
				trait = craven
				trait = cruel
			}
		}

		add_trait_silently_kind_effect = yes
	}

	option = {
		name = EVTOPTD36000

		trigger = {
			NOR = {
				trait = craven
				trait = humble
				trait = content
			}
		}

		if = {
			limit = { martial < 10 }

			random_list = {
				25 = { }

				15 = {
					if = {
						limit = { trait = diligent }
						prestige = 10
					}
					else = {
						tooltip = {
							remove_trait_silently_slothful_effect = yes
							add_trait = diligent
						}

						hidden_effect = {
							remove_trait_silently_slothful_effect = yes
							add_trait_diligent_effect = yes
						}
					}
				}

				30 = { change_martial = 1 }
				20 = { add_trait_wounded_effect = yes }
				8 = { add_maimed_trait_effect = yes }
				2 = { death = { death_reason = death_accident } }
			}
		}

		else_if = {
			limit = { martial < 20 }

			random_list = {
				50 = { }

				10 = {
					if = {
						limit = { trait = diligent }
						prestige = 10
					}
					else = {
						tooltip = {
							remove_trait_silently_slothful_effect = yes
							add_trait = diligent
						}

						hidden_effect = {
							remove_trait_silently_slothful_effect = yes
							add_trait_diligent_effect = yes
						}
					}
				}

				20 = { change_martial = 1 }
				15 = { add_trait_wounded_effect = yes }
				4 = { add_maimed_trait_effect = yes }
				1 = { death = { death_reason = death_accident } }
			}
		}

		else = {
			random_list = {
				80 = { }

				13 = {
					if = {
						limit = { trait = diligent }
						prestige = 10
					}
					else = {
						tooltip = {
							remove_trait_silently_slothful_effect = yes
							add_trait = diligent
						}

						hidden_effect = {
							remove_trait_silently_slothful_effect = yes
							add_trait_diligent_effect = yes
						}
					}
				}

				2 = { add_maimed_trait_effect = yes }

	has_character_flag = hunt_begin
	min_age = 16
	only_capable = yes
	prisoner = no

			flag = hunt_begin
			days >= 3
		}

		NOT = { has_character_flag = hunt_paranoia }

		any_courtier = {
			has_plot = plot_kill_character

			plot_target_char = {
				character = ROOT
			}
		}

			limit = { has_character_flag = hunt_event_happened }
