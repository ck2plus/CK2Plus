# Ash'ari liege has Mu'tazila vassal (Hidden)
character_event = {
	id = SoA.602

	hide_window = yes

	only_rulers = yes
	religion_group = muslim
	has_dlc = "Sons of Abraham"
	min_age = 16
	only_capable = yes
	prisoner = no

	trigger = {
		OR = {
			religion = sunni
			religion = zikri
			religion = quranist
		}

		trait = mutazilite
		is_inaccessible_trigger = no

		liege = {
			trait = ashari
			war = no

			OR = {
				religion = sunni
				religion = zikri
				religion = quranist
			}

			NOT = {
				has_opinion_modifier = {
					name = opinion_rejected_school
					who = ROOT
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 640
	}

	immediate = {
		liege = {
			character_event = { id = SoA.603 }
		}
	}
}

		}

		name = EVTOPTB_SoA_604

		FROM = {
			show_scope_change = no

				name = opinion_rejected_school
				who = ROOT
			}

			letter_event = {
				id = SoA.606
				days = 3

			}
		}
	}
}
