			}

			save_event_target_as = spiritual_gender
		}

			50 = {

					limit = { is_female = no }

		name = EVTOPTA_MNM_3420

				factor = 5
				trait = just
			}

				factor = 5
				trait = trusting
			}
		}
	}

	option = {
		name = EVTOPTB_MNM_3420

		custom_tooltip = { text = EVTOPTB_MNM_3420_TT }

		event_target:secret_religion_recruiter = {
