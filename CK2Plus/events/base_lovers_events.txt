# Confront cheating spouse/spouse's lover
character_event = {
	id = 64000
	desc = EVTDESC64000
	picture = GFX_evt_quarrel

	is_triggered_only = yes # on_yearly_pulse, random_events, base_lover_events

	lacks_dlc = "Way of Life"
	is_married = yes
	only_capable = yes
	prisoner = no

	trigger = {
		spouse = {
			has_lover = yes
			in_command_trigger = no
			war = no
			is_abroad = no
			prisoner = no
			NOT = { is_lover = ROOT }

			any_lover = {
				NOR = {
					is_married = PREV
					is_consort = PREV
					PREV = { is_consort = PREV }
				}
			}
		}
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2
			trait = envious
		}
	}

	immediate = {
		spouse = {
			random_lover = {
				limit = {
					NOR = {
						is_married = PREV
						is_consort = PREV
						PREV = { is_consort = PREV }
					}
				}

				save_event_target_as = target_lover
			}
		}
	}

	option = {
		name = EVTOPTA64000

		spouse = {
			show_scope_change = no

			character_event = {
				id = 64001
				tooltip = EVTTOOLTIP64001
			}
		}
	}

	option = {
		name = EVTOPTB64000

		event_target:target_lover = {
			show_scope_change = no

			character_event = {
				id = 64002
				tooltip = EVTTOOLTIP64002
			}
		}
	}
}

		factor = 1

				event_target:target_lover = { is_female = no }
			}
		}

				event_target:target_lover = { is_female = yes }
