# If in the red => increase revolt risk in a province.
province_event = {
	id = 68000
	desc = EVTDESC68000
	picture = GFX_evt_bandits

	notification = yes

	trigger = {
		owner = { wealth < 0 }
		NOT = { has_province_modifier = incompetent_rule }
	}

	mean_time_to_happen = {
		years = 20

		# Lower if really badly in debt
		mult_modifier = {
			factor = 0.9

			owner = {
				NOT = {
					scaled_wealth = {
						value = -0.25
						min = -50
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.7

			owner = {
				NOT = {
					scaled_wealth = {
						value = -0.5
						min = -100
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.5

			owner = {
				NOT = {
					scaled_wealth = {
						value = -1
						min = -200
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.3

			owner = {
				NOT = {
					scaled_wealth = {
						value = -2.5
						min = -300
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.1

			owner = {
				NOT = {
					scaled_wealth = {
						value = -5
						min = -500
					}
				}
			}
		}

		# Traits affect this
		mult_modifier = {
			factor = 1.5
			owner = { trait = ambitious }
		}

		mult_modifier = {
			factor = 5.0
			owner = { trait = just }
		}

		mult_modifier = {
			factor = 2.0
			owner = { trait = midas_touched }
		}

		mult_modifier = {
			factor = 0.5
			owner = { trait = arbitrary }
		}

		mult_modifier = {
			factor = 0.7
			owner = { trait = greedy }
		}

		mult_modifier = {
			factor = 0.7
			owner = { trait = cruel }
		}

		mult_modifier = {
			factor = 0.1
			owner = { trait = slothful }
		}

		mult_modifier = {
			factor = 0.2
			owner = { is_incapable = yes }
		}

		mult_modifier = {
			factor = 0.1
			owner = { trait = infirm }
		}

		# Modify by stewardship and intrigue
		mult_modifier = {
			factor = 0.7
			owner = { realm_stewardship < 15 }
		}

		mult_modifier = {
			factor = 0.7
			owner = { realm_stewardship < 10 }
		}

		mult_modifier = {
			factor = 0.7
			owner = { realm_stewardship < 5 }
		}

		mult_modifier = {
			factor = 2.0
			owner = { realm_stewardship >= 25 }
		}

		mult_modifier = {
			factor = 1.5
			owner = { realm_stewardship >= 20 }
		}

		mult_modifier = {
			factor = 2.0
			owner = { realm_intrigue >= 25 }
		}

		mult_modifier = {
			factor = 1.5
			owner = { realm_intrigue >= 20 }
		}
	}

	immediate = {
		add_province_modifier = {
			name = incompetent_rule
			duration = -1
		}
	}

	option = {
		name = EVTOPTA68000
	}
}
