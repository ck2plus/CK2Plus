# Regent embezzles money
character_event = {
	id = 61000
	desc = EVTDESC61000

	picture = GFX_evt_council
	border = GFX_event_normal_frame_economy

	min_age = 16
	prisoner = no
	capable_only = yes

	trigger = {
		has_minor_title = title_regent
		independent = no
		liege = { wealth = 20 }
		OR = {
			trait = greedy
			trait = deceitful
			trait = envious
			trait = arbitrary
			trait = ambitious
		}
		NOR = {
			trait = charitable
			trait = honest
			trait = just
		}
	}

	mean_time_to_happen = {
		months = 280
		modifier = {
			factor = 0.5
			trait = greedy
		}
		modifier = {
			factor = 0.75
			trait = deceitful
		}
		modifier = {
			factor = 0.5
			trait = envious
		}
		modifier = {
			factor = 0.75
			trait = arbitrary
		}
		modifier = {
			factor = 2.0
			trait = kind
		}
		modifier = {
			factor = 2.0
			trait = patient
		}
	}

	option = {
		name = EVTOPTA61000 # Embezzle
		ai_chance = { factor = 100 }
		liege = { wealth = -20 }
		wealth = 20
		random_list = {
			50 = {
				modifier = {
					factor = 2.0
					has_job_title = job_treasurer
				}
				# Got away with it!
			}
			40 = {
				liege = { character_event = { id = 61001 days = 4 tooltip = EVTTOOLTIP61001 } } # Liege suspects
				if = {
					limit = { NOT = { has_job_title = job_treasurer }  }
					liege = { job_treasurer = { character_event = { id = 61002 days = 4 tooltip = EVTTOOLTIP61002 } } } # The Steward suspects
				}
			}
			10 = {
				liege = { character_event = { id = 61003 days = 4 tooltip = EVTTOOLTIP61003 } } # Liege has proof
				if = {
					limit = { NOT = { has_job_title = job_treasurer }  }
					liege = { job_treasurer = { character_event = { id = 61004 days = 4 tooltip = EVTTOOLTIP61004 } } } # The Steward has proof
				}
			}
		}
	}

	option = {
		name = EVTOPTB61000 # I dare not!
		ai_chance = { factor = 0 }
		piety = 1
	}
}

			factor = 0.5

					can_be_given_away = yes

						tier = DUKE
						tier = KING
						tier = EMPEROR
						location = { is_capital = yes }
					}
					ROOT = { has_claim = PREV }
				}
			}
		}
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.75
			trait = deceitful
		}
		modifier = {
			factor = 0.5
			trait = envious
		}
		modifier = {
			factor = 0.75
			trait = arbitrary
		}
		modifier = {
			factor = 2.0
			trait = kind
		}

			factor = 2.0
			trait = just
		}
	}

	option = {
		name = EVTOPTA61100 #Grab holding.
		ai_chance = {
			factor = 90
			modifier = {

					trait = just
					trait = content

					}
				}
				NOT = { trait = ambitious }
			}
			modifier = {
				factor = 0.5
				trait = kind
				NOT = { trait = ambitious }
			}
			modifier = {
				factor = 0.5
				trait = charitable
				NOT = { trait = ambitious }
			}
		}

			character_event = {	id = 61101 tooltip = EVTTOOLTIP61101 }

		name = EVTOPTB61100 #Do not grab holding.
		ai_chance = {
			factor = 10
		}
		ROOT = { piety = 40 }
	}
}

		}
