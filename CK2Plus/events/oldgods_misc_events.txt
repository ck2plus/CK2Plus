		}

		if = {
			limit = {
				NOT = {
					any_war = {
						any_attacker = { character = FROM }
					}
				}
			}

			piety = -50

				show_scope_change = no

				limit = { religion = ROOT }

					name = opinion_failed_invasion

				}
			}
		}
	}
}
