	desc = EVTDESC_TOG_100

	hide_from = yes

	option = {
		name = EVTOPTA_TOG_100

		custom_tooltip  = { text = EVTTOOLTIP_TOG_101 }

		any_vassal = {
			show_scope_change = no

			limit = {
				is_playable = yes
				block_general_event_trigger = no
				practical_age < 55
				trigger_if = {
					limit = { is_female = yes }
					trait = martial_lady_tengri
				}
				war = no
				is_maimed_trigger = no

				NOR = {
					trait = wounded
					trait = infirm
				}
			}

			set_character_flag = do_not_disturb

			character_event = { id = TOG.101 }

			character_event = {
				id = TOG.299 # Safety catch flag clearing
				days = 100
			}

			character_event = {
				id = TOG.103 # Hunt begins
				days = 10
			}
		}

		character_event = {
			id = TOG.102 # Hunt begins
			days = 10
		}
	}
}
