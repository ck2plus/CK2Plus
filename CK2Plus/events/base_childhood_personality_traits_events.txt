# Event ID 1500-1509 is reserved for Gluttonous events

# Gain Gluttony
character_event = {
	id = 1500
	desc = EVTDESC1500
	picture = GFX_evt_child_play

	is_triggered_only = yes # on_yearly_childhood_pulse, random_events

	lacks_dlc = "Conclave"
	prisoner = no
	only_capable = yes

	trigger = {
		has_guardian = yes
		personality_traits < 5

		NOR = {
			trait = temperate
			trait = gluttonous
			has_character_flag = is_child_of_destiny
		}
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2
			trait = slothful
		}

		mult_modifier = {
			factor = 1.05
			trait = clubfooted
		}

		mult_modifier = {
			factor = 1.10
			trait = harelip
		}

		mult_modifier = {
			factor = 1.5
			trait = hunchback
		}

		mult_modifier = {
			factor = 1.05
			trait = lisp
		}

		mult_modifier = {
			factor = 1.05
			trait = stutter
		}

		mult_modifier = {
			factor = 0.95
			has_fair_trait_trigger = yes
		}

		mult_modifier = {
			factor = 1.10
			has_ugly_trait_trigger = yes
		}

		mult_modifier = {
			factor = 1.2
			trait = dwarf
		}

		mult_modifier = {
			factor = 0.95
			trait = genius
		}

		mult_modifier = {
			factor = 0.95

			OR = {
				trait = quick
				trait = shrewd
			}
		}

		mult_modifier = {
			factor = 1.2

			OR = {
				trait = slow
				trait = dull
			}
		}

		mult_modifier = {
			factor = 1.5
			trait = imbecile
		}

		mult_modifier = {
			factor = 1.5
			trait = inbred
		}

		mult_modifier = {
			factor = 0.05
			is_strong_trigger = yes
		}

		mult_modifier = {
			factor = 5.0
			is_weak_trigger = yes
		}

		# CK2Plus
		mult_modifier = {
			factor = 2
			guardian = { trait = gluttonous }
		}

		mult_modifier = {
			factor = 0.5
			guardian = { trait = temperate }
		}
	}

	option = {
		name = EVTOPTA1500

		add_trait_silently_gluttonous_effect = yes

		guardian = {
			character_event = {
				id = 2000
				days = 3
			}
		}
	}
}

# Event ID 2000-2009 is reserved for Gluttonous events #
# Fosterling gains trait Gluttonous
character_event = {
	id = 2000
	desc = EVTDESC2000
	picture = GFX_evt_child_play

	is_triggered_only = yes

	option = {
		name = EVTOPTA2000 # Encourage Gluttony

		trigger_switch = {
			on_trigger = trait

			temperate = { prestige = -5 }
			gluttonous = { prestige = 5 }
		}

		FROM = {
			show_scope_change = no

			tooltip = {
				add_trait = gluttonous
			}

			opinion = {
				name = opinion_happy_with_guardian
				who = ROOT
				years = 5
			}
		}

		ai_chance = {
			factor = 10

			trigger = {
				NOT = { trait = temperate }
			}

			mult_modifier = {
				factor = 4
				trait = gluttonous
			}

			mult_modifier = {
				factor = 2
				trait = slothful
			}

			mult_modifier = {
				factor = 2
				trait = hedonist
			}
		}
	}

	option = {
		name = EVTOPTB2000 # Birch Him!

		trigger = {
			NOT = { trait = kind }
		}

		pacifists_lose_piety_effect = yes

		reverse_opinion = {
			name = opinion_sad
			who = FROM
			years = 5
		}

		random = {
			chance = 40

			FROM = {
				show_scope_change = no

				tooltip = {
					remove_trait = gluttonous
				}

				character_event = { id = 1501 }
			}
		}

		ai_chance = {
			factor = 10

			mult_modifier = {
				factor = 5
				trait = cruel
			}

			mult_modifier = {
				factor = 5
				trait = wroth
			}

			mult_modifier = {
				factor = 0.1
				trait = gluttonous
			}
		}
	}

	option = {
		name = EVTOPTC2000 # Talk to Him about moderation.

		trigger = {
			NOT = { trait = gluttonous }
		}

		FROM = {
			show_scope_change = no

			random_list = {
				50 = {
					add_trait_fully_temperate_effect = yes

					ROOT = {
						character_event = { id = 38176 } # Notify guardian
					}
				}

				50 = {
					tooltip = {
						remove_trait = gluttonous
					}

					character_event = { id = 1501 }
				}
			}
		}

		ai_chance = {
			factor = 10

			mult_modifier = {
				factor = 20
				trait = temperate
			}

			mult_modifier = {
				factor = 10
				trait = diligent
			}

			mult_modifier = {
				factor = 10
				trait = celibate
			}
		}
	}

	option = {
		name = EVTOPTD2000 # Pray God will save her..

		trigger = {
			OR = {
				religion_group = christian
				religion_group = muslim
				religion_group = jewish_group
			}

			NOT = { trait = cynical }
		}

		piety = 5

		random = {
			chance = 20

			FROM = {
				show_scope_change = no

				tooltip = {
					remove_trait = gluttonous
				}

				character_event = { id = 1501 }
			}
		}

		ai_chance = {
			factor = 2

			mult_modifier = {
				factor = 150
				trait = zealous
			}
		}
	}
}

# Event ID 2010-2019 is reserved for Temperate events
# Fosterling is Temperate
character_event = {
	id = 2010
	desc = EVTDESC2010
	picture = GFX_evt_child_play

	is_triggered_only = yes

	option = {
		name = EVTOPTA2010 # Encourage

		trigger_switch = {
			on_trigger = trait

			gluttonous = { prestige = -5 }
			temperate = { prestige = 5 }
		}

		if = {
			limit = { stewardship >= 10 }

			FROM = {
				show_scope_change = no

				change_stewardship = 1
			}
		}

		FROM = {
			show_scope_change = no

			tooltip = {
				add_trait = temperate
			}

			opinion = {
				name = opinion_happy_with_guardian
				who = ROOT
				years = 5
			}
		}

		ai_chance = {
			factor = 10

			mult_modifier = {
				factor = 5
				trait = temperate
			}

			mult_modifier = {
				factor = 4
				trait = diligent
			}

			mult_modifier = {
				factor = 4
				trait = celibate
			}
		}
	}

	option = {
		name = EVTOPTB2010 # Hold up X as an example for the court!

		trigger = {
			any_courtier = {
				trait = gluttonous
			}

			NOT = { trait = gluttonous }
		}

		if = {
			limit = { stewardship >= 10 }

			FROM = {
				show_scope_change = no

				change_stewardship = 1
			}
		}

		any_courtier = {
			show_scope_change = no

			limit = { trait = gluttonous }

			opinion = {
				name = opinion_envy
				who = FROM
				years = 5
			}
		}

		FROM = {
			show_scope_change = no

			prestige = 5
		}

		ai_chance = {
			factor = 10

			mult_modifier = {
				factor = 5
				trait = temperate
			}

			mult_modifier = {
				factor = 10
				trait = proud
			}
		}
	}

	option = {
		name = EVTOPTC2010 # Thank God for X's virtues!

		trigger = {
			NOR = {
				trait = cynical
				religion = buddhist
				religion = jain
			}
		}

		piety = 5

		random = {
			chance = 10

			FROM = {
				show_scope_change = no
				add_trait_fully_gluttonous_effect = yes
			}

			character_event = { id = 38177 } # Notify guardian
		}

		ai_chance = {
			factor = 2

			mult_modifier = {
				factor = 150
				trait = zealous
			}
		}
	}

	option = {
		name = EVTOPTD2010 # He/She must learn to enjoy life!

		trigger = {
			NOT = { trait = temperate }
		}

		FROM = {
			show_scope_change = no

			random_list = {
				50 = {
					add_trait_fully_gluttonous_effect = yes

					ROOT = {
						character_event = { id = 38177 } # Notify guardian
