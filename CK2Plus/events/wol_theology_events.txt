					name = religious_philosopher

# Embrace heresy?
character_event = {
	id = WoL.7060
	desc = EVTDESC_WoL_7060
	picture = GFX_evt_religious_exultation
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes # on_focus_pulse, random_events

	only_playable = yes
	min_age = 16
	only_capable = yes

	trigger = {
		has_focus = focus_theology
		trait = zealous
		has_heresies = yes
		can_change_religion = yes

		OR = {
			religion_group = christian
			religion_group = muslim
			religion_group = zoroastrian_group
			religion_group = jewish_group
		}

		is_heretic = no
		is_priest = no
		holy_order = no
		mercenary = no
		controls_religion = no

		NOR = {
			has_character_flag = wol_theology_heretic
			rightful_religious_head = ROOT
		}
	}

	weight_multiplier = {
		days = 1

		mult_modifier = {
			factor = 0.5
			higher_real_tier_than = DUKE
		}

		mult_modifier = {
			factor = 0.5
			higher_real_tier_than = KING
		}
	}

	immediate = {
		set_character_flag = wol_theology_heretic
		set_character_flag = convert_heretic
	}

	option = {
		name = EVTOPTA_WoL_7060 # Settle firmly into orthodoxy

		piety = 30

		ai_chance = { factor = 40 }
	}

	option = {
		name = EVTOPTB_WoL_7060 # Take middle ground

		remove_trait = zealous

		ai_chance = { factor = 50 }
	}

	option = {
		name = EVTOPTC_WoL_7060 # Become heretic

		piety = 100

		if = {
			limit = { ai = no }

			# Workaround for automatic usurpation of theocratic characters
			religion_authority = { name = ruler_converted_from }
			convert_religion_failsafe_effect = yes
			religion_authority = { name = ruler_converted_to }

			hidden_effect = {
				any_courtier = {
					limit = {
						dynasty = ROOT
						is_adult = no
						is_ruler = no
					}

					religion = ROOT
				}
			}

			set_character_flag = converted_religion
		}
		else = {
			character_event = { id = WoL.7061 }
		}

		if = {
			limit = {
				NOR = {
					has_character_modifier = religious_philosopher
					trait = theologian
				}
			}

			if = {
				limit = {
					NOT = { has_character_modifier = religious_studies }
				}

				add_character_modifier = {
					name = religious_studies
					duration = -1
				}

				set_character_flag = religious_studies_timer
			}
			else_if = {
				limit = {
					had_character_flag = {
						flag = religious_studies_timer
						days >= 1800
					}
				}

				clr_character_flag = religious_studies_timer
				remove_character_modifier = religious_studies

				add_character_modifier = {
					name = religious_philosopher
					duration = -1
				}
			}
		}

		ai_chance = { factor = 10 }
	}

	after = {
		clr_character_flag = convert_heretic
	}
}
