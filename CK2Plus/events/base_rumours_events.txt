# Suspected Sodomite: Any Courtier
character_event = {
	id = 69000
	desc = EVTDESC69000
	picture = GFX_evt_whispers

	is_triggered_only = yes # on_yearly_pulse, random_events

	min_age = 16
	only_capable = yes
	prisoner = no

	trigger = {
		intrigue >= 10
		is_ruler = no
		NOT = { has_character_flag = rumour_spreader }

		any_courtier = {
			is_female = no
			is_adult = yes

			OR = {
				martial < 2
				trait = craven
				trait = poet
				trait = gardener
			}

			reverse_opinion = {
				who = ROOT
				value < 50
			}

			prisoner = no
			has_children = no

			NOR = {
				has_character_flag = suspected_sodomite
				trait = homosexual
			}
		}
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2
			trait = deceitful
		}

		mult_modifier = {
			factor = 2
			intrigue >= 15
		}

		mult_modifier = {
			factor = 2
			intrigue >= 20
		}
	}

	immediate = {
		set_character_flag = rumour_spreader

		random_courtier = {
			show_scope_change = no

			limit = {
				is_female = no
				is_adult = yes

				OR = {
					martial < 2
					trait = craven
					trait = poet
					trait = gardener
				}

				reverse_opinion = {
					who = ROOT
					value < 50
				}

				prisoner = no
				has_children = no

				NOR = {
					trait = homosexual
					has_character_flag = suspected_sodomite
				}
			}

			save_event_target_as = sodomite_target
		}
	}

	option = {
		name = EVTOPTA69000

		event_target:sodomite_target = {
			show_scope_change = no

			character_event = {
				id = 69001
				days = 3
				tooltip = EVTTOOLTIP69001
			}
		}

		ai_chance = { factor = 10 }
	}

	option = {
		name = EVTOPTB69000

		event_target:sodomite_target = {
			show_scope_change = no

			set_character_flag = suspected_sodomite

			character_event = {
				id = 69080
				days = 3
				tooltip = EVTTOOLTIP69080
			}
		}

		ai_chance = { factor = 50 }
	}

	option = {
		name = EVTOPTC69000

		event_target:sodomite_target = {
			show_scope_change = no

			character_event = {
				id = 69002
				days = 3
				tooltip = EVTTOOLTIP69002
			}
		}

		ai_chance = { factor = 40 }
	}
}
