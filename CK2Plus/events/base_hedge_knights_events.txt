# A young woman has been seduced by a hedge knight
character_event = {
	id = 71004
	desc = EVTDESC71004
	picture = GFX_evt_lovers

	ai = no
	has_character_flag = hedge_knights_visiting
	min_age = 16
	prisoner = no

	trigger = {
		any_courtier = {
	 		is_female = yes
	 		is_adult = yes
	 		practical_age < 20
			fertility >= 0.3
	 		is_married = no
	 		is_pregnant = no
			has_lover = no
	 	}

		NOR = {
			has_character_flag = hedge_knight_recruited
			has_character_flag = hedge_knight_woman_seduced
		}
	}

	immediate = {
		set_character_flag = hedge_knight_woman_seduced

		create_character = {
			random_traits = yes
			dynasty = random
			fertility = 0.6
			female = no
			age = 24
			culture = ROOT
			religion = ROOT
	 	}

		new_character = {
			save_event_target_as = hedge_knight
		}

		any_courtier = {
	 		limit = {
	 			is_female = yes
	 			is_adult = yes
	 			practical_age < 20
				fertility >= 0.3
	 			is_married = no
	 			is_pregnant = no
				has_lover = no
	 		}

	 		score_value = {
				value = 10
				interesting_courtier_selection_score = yes
			}

	 		save_event_target_as = seduced_courtier
	 	}
	}

	mean_time_to_happen = {
		days = 20

		mult_modifier = {
			factor = 5
			has_character_flag = hedge_knight_recruited
		}
	}

	option = {
		name = EVTOPTA71004 # Stay

		event_target:seduced_courtier = {
			show_scope_change = no

	 		add_trait_lustful_effect = yes
		}

		event_target:hedge_knight = {
			show_scope_change = no

			add_trait = lustful
	 		impregnate_cuckoo = event_target:seduced_courtier
	 	}

	 	piety = -10
	}

	option = {
		name = EVTOPTB71004 # Dungeon

		ai_chance = {
			factor = 1

			mult_modifier = {
				factor = 100

				OR = {
					trait = cruel
					trait = zealous
				}
			}
		}

		event_target:seduced_courtier = {
			show_scope_change = no

	 		add_trait_lustful_effect = yes
		}

		event_target:hedge_knight = {
			show_scope_change = no

			add_trait = lustful
	 		impregnate_cuckoo = event_target:seduced_courtier
			imprison = ROOT
	 	}
	}
}
