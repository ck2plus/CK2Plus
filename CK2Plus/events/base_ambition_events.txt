# Toy Army
character_event = {
	id = ambition.001
	desc = evt_ambition_001_desc_a
	picture = GFX_evt_large_army
	border = GFX_event_normal_frame_war

	is_triggered_only = yes # on_yearly_pulse, random_events, ambition_events

	only_playable = yes
	only_men = yes
	only_capable = yes
	is_married = yes
	prisoner = no

	trigger = {
		always = no # Disabled in CK2Plus

		OR = {
			has_ambition = obj_improve_martial

			AND = {
				has_focus = focus_war
				martial < 8
			}
		}

		NOR = {
			has_character_flag = toy_army
			has_character_flag = toy_army_taken
			has_character_flag = street_ruffians
			has_character_flag = warrior_mentor
		}

		is_inaccessible_trigger = no
	}


	option = {
		name = evt_ambition_001_opt_a

		change_martial = 1
		set_character_flag = toy_army
		set_character_flag = toy_army_prepare
		set_character_flag = toy_army_taken
	}
}

### Alexander/Ashoka the Great
character_event = {
	id = 94018
	picture = GFX_evt_library
	border = GFX_event_normal_frame_war

	desc = {
		text = EVTDESC94018
		trigger = {
			capital_scope = {
				NOT = { region = world_india }
			}
		}
	}
	desc = {
		text = EVTDESC94018_india
		trigger = {
			capital_scope = {
				region = world_india
			}
		}
	}

	is_triggered_only = yes # on_yearly_pulse, random_events, ambition_events

	only_playable = yes
	min_age = 16
	only_capable = yes
	prisoner = no

	trigger = {
		OR = {
			has_ambition = obj_improve_martial

			trigger_if = {
				limit = { has_focus = focus_war }

				OR = {
					lifestyle_traits < 1

					trigger_if = {
						limit = { martial < 8 }

						NOR = {
							has_character_flag = toy_army
							has_character_flag = street_ruffians
							has_character_flag = warrior_mentor
							trait = strategist
						}
					}
				}

				NOR = {
					has_character_modifier = student_of_strategy
					has_character_modifier = adept_of_strategy
				}
			}
		}
	}

	option = {
		name = {
			text = EVTOPTA94018
			trigger = {
				capital_scope = {
					NOT = { region = world_india }
				}
			}
		}
		name = {
			text = EVTOPTB_94018
			trigger = {
				capital_scope = {
					region = world_india
				}
			}
		}

		trigger = {
			OR = {
				has_ambition = obj_improve_martial
				martial < 8
			}
		}

		change_martial = 1
	}

	option = {
		name = EVTOPTC_94018 # Study further

		trigger = { has_focus = focus_war }

		add_character_modifier = {
			name = student_of_strategy
			duration = -1
		}

		character_event = {
			id = 94019
			days = 300
			random = 900
			tooltip = studying_strategy_tooltip
		}
	}
}
