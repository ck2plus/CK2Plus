character_event = {
	id = GoPagan.120
	
	hide_window = yes
	
	min_age = 16
	capable_only = yes
	prisoner = no
	ai = yes

	trigger = {
		is_heretic = no
		piety = 0
		health = 3
		is_priest = no
		controls_religion = no
		OR = {
			NOT = { religion_authority = 0.20 }
			AND = {
				NOT = { religion_authority = 0.40 }
				trait = lunatic
			}
		}
		NOT = {
			has_character_flag = can_go_pagan
			has_character_flag = gone_pagan
			trait = zealous
			trait = crusader
			trait = mujahid
			trait = holy_warrior
			has_character_modifier = voice_of_jesus
			has_character_modifier = voice_of_satan
			piety = 500
			religion = monophysite
			religion = miaphysite
			religion = pagan
			AND = {
				religion_group = pagan_group
				is_reformed_religion = no
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.25
			trait = lunatic
		}
		modifier = {
			factor = 0.25
			trait = cynical
		}
		modifier = {
			factor = 0.25
			has_character_modifier = defied_church
		}
		modifier = {
			factor = 0.75
			NOT = { religion_authority = 0.15 }
		}
		modifier = {
			factor = 0.5
			NOT = { religion_authority = 0.10 }
		}
		modifier = {
			factor = 0.5
			trait = mystic
		}
		modifier = {
			factor = 0.5
			location = { religion_group = pagan_group }
		}
		modifier = {
			factor = 0.5
			independent = no
			liege = { religion_group = pagan_group }
		}
		modifier = {
			factor = 0.5
			OR = {
				trait = sympathy_pagans
				trait = sympathy_zoroastrianism
				trait = sympathy_judaism
				trait = sympathy_indian
			}
		}
		modifier = {
			factor = 0.75
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 0.75
			NOT = { learning = 2 }
		}

			factor = 3

		}
		modifier = {
			factor = 3
			trait = content
		}
		modifier = {
			factor = 3
			trait = craven
		}
		modifier = {
			factor = 3
			trait = patient
		}
		modifier = {
			factor = 3

		}
