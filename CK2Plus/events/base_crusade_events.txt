# Conquers the crusade target
# on_siege_won_leader

	id = 8330
	desc = EVTDESC8330
	picture = GFX_evt_siege
	border = GFX_event_normal_frame_war

	is_triggered_only = yes

	trigger = {
		has_called_crusade = yes
		FROM = { is_capital = yes }

		OR = {
			any_war = {
				attacker = { religion = ROOT }

				OR = {
					using_cb = crusade
					using_cb = new_crusade
				}
			}

			any_liege = {
				any_war = {
					attacker = { religion = ROOT }

					OR = {
						using_cb = crusade
						using_cb = new_crusade
					}
				}
			}
		}

		crusade_target = {
			de_jure_vassal_or_below = FROM
		}
	}

	option = {
		name = EVTOPTA8330

		prestige = 100
		piety = 100

		set_character_flag = captures_crusade_target

		if = {
			limit = { independent = no }

			# Inform liege
			if = {
				limit = {
					liege = {
						independent = no # isn't top liege

						any_war = {
							OR = {
								using_cb = crusade
								using_cb = new_crusade
							}
						}
					}
				}

				liege = {
					letter_event = {
						id = 8331
						days = 1
					}
				}
			}

			# Inform top liege
			if = {
				limit = {
					top_liege = {
						any_war = {
							OR = {
								using_cb = crusade
								using_cb = new_crusade
							}
						}
					}
				}

				top_liege = {
					letter_event = {
						id = 8331
						days = 1
					}
				}
			}
		}
	}
}

	id = 8400
	desc = EVTDESC8400

	only_playable = yes
	only_men = yes
	only_capable = yes
	min_age = 16
	prisoner = no

	is_triggered_only = yes # on_yearly_pulse

	trigger = {
		controls_religion = yes
		has_called_crusade = yes

		trigger_if = {
			limit = { uses_new_crusade = no }

			has_called_crusade = yes

			had_global_flag = {
				flag = crusade_called
				months >= 6
			}
		}
		trigger_else = {
			is_preparing_crusade = yes
		}

		any_independent_ruler = {
			religion = ROOT
			rightful_religious_head = ROOT # filters out rulers with antipopes

				any_war = {
					OR = {
						using_cb = crusade
						using_cb = new_crusade
					}
				}
			}

			crusade_target = {
				holder_scope = {
					NOR = {
						reverse_has_truce = PREVPREV
						has_non_aggression_pact_with = PREVPREV
					}
				}
			}

			NOR = {
				excommunicated_for = ROOT

				reverse_has_opinion_modifier = {
					who = ROOT
					name = opinion_not_participating_in_crusade_emperor
				}

				reverse_has_opinion_modifier = {
					who = ROOT
					name = opinion_not_participating_in_crusade_king
				}

				reverse_has_opinion_modifier = {
					who = ROOT
					name = opinion_not_participating_in_crusade_duke
				}

				reverse_has_opinion_modifier = {
					who = ROOT
					name = opinion_not_participating_in_crusade_count
				}
			}

			NOT = {
				any_realm_province = {
					any_neighbor_province = {
						is_land = yes
						NOT = { religion_group = christian }
					}
				}
			}
		}
	}

	weight_multiplier = {
		days = 1

		mult_modifier = {
			factor = 0.5
			trait = zealous
		}
	}

	option = {
		name = EVTOPTA8400

		random_independent_ruler = {
			limit = {
				religion = ROOT
				rightful_religious_head = ROOT # filters out rulers with antipopes
				has_pledged_crusade_participation = no
				mercenary = no

				NOT = {
					any_war = {
						OR = {
							using_cb = crusade
							using_cb = new_crusade
						}
					}
				}

				crusade_target = {
					holder_scope = {
						NOR = {
							reverse_has_truce = PREVPREV
							has_non_aggression_pact_with = PREVPREV
						}
					}
				}

				NOR = {
					excommunicated_for = ROOT

					reverse_has_opinion_modifier = {
						who = ROOT
						name = opinion_not_participating_in_crusade_emperor
					}

					reverse_has_opinion_modifier = {
						who = ROOT
						name = opinion_not_participating_in_crusade_king
					}

					reverse_has_opinion_modifier = {
						who = ROOT
						name = opinion_not_participating_in_crusade_duke
					}

					reverse_has_opinion_modifier = {
						who = ROOT
						name = opinion_not_participating_in_crusade_count
					}

						any_neighbor_province = {
